## 6. NUOVA STRUTTURA DATABASE

### Tabella `daily_habits`
```sql
CREATE TABLE daily_habits (
  id UUID PRIMARY KEY,
  user_id UUID NOT NULL,
  date DATE NOT NULL,
  habit_type TEXT NOT NULL, -- 'water', 'cigarettes', 'exercise', 'meditation', etc.
  value NUMERIC, -- Litri, numero sigarette, minuti, etc.
  unit TEXT, -- 'L', 'pieces', 'minutes', 'steps'
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(user_id, date, habit_type)
);
```

### Tabella `body_metrics`
```sql
CREATE TABLE body_metrics (
  id UUID PRIMARY KEY,
  user_id UUID NOT NULL,
  date DATE NOT NULL,
  weight NUMERIC,
  waist_circumference NUMERIC,
  sleep_hours NUMERIC,
  resting_heart_rate INTEGER,
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(user_id, date)
);
```

### Tabella `achievements`
```sql
CREATE TABLE user_achievements (
  id UUID PRIMARY KEY,
  user_id UUID NOT NULL,
  achievement_id TEXT NOT NULL, -- 'first_checkin', 'week_streak', etc.
  unlocked_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(user_id, achievement_id)
```
